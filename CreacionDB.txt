-- ============================================================

CREATE DATABASE IF NOT EXISTS novarider;
USE novarider;

-- ============================================================
-- 1. SEGURIDAD Y USUARIOS
-- ============================================================

CREATE TABLE TRoles (
    id_rol INT PRIMARY KEY AUTO_INCREMENT,
    nombre VARCHAR(50) NOT NULL UNIQUE,
    descripcion VARCHAR(255),

    estadoA BOOLEAN DEFAULT TRUE,
    usuarioA INT,
    fechahoraA DATETIME DEFAULT CURRENT_TIMESTAMP
);

-- TPersonas ELIMINADA - sus campos se fusionaron en TEmpleados y TClientes

CREATE TABLE TEmpleados (
    id_empleado INT PRIMARY KEY AUTO_INCREMENT,
    -- Datos de persona (antes en TPersonas)
    ci VARCHAR(255),
    primer_nombre VARCHAR(255),
    segundo_nombre VARCHAR(255),
    apellido_paterno VARCHAR(255),
    apellido_materno VARCHAR(255),
    fecha_nacimiento DATE,
    telefono VARCHAR(255),
    -- Datos de empleado
    fecha_ingreso DATE,
    sueldo_base DECIMAL DEFAULT 0,
    cargo VARCHAR(255),

    estadoA BOOLEAN DEFAULT TRUE,
    usuarioA INT,
    fechahoraA DATETIME DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE TUsuarios (
    id_usuario INT PRIMARY KEY AUTO_INCREMENT,
    id_empleado INT UNIQUE,
    username VARCHAR(255) NOT NULL,
    password_hash VARCHAR(255) NOT NULL,

    estadoA BOOLEAN DEFAULT TRUE,
    usuarioA INT,
    fechahoraA DATETIME DEFAULT CURRENT_TIMESTAMP,

    FOREIGN KEY (id_empleado) REFERENCES TEmpleados(id_empleado)
);

-- Tabla pivote para roles M:N
CREATE TABLE TUsuarioRol (
    id_usuario INT NOT NULL,
    id_rol INT NOT NULL,
    PRIMARY KEY (id_usuario, id_rol),
    FOREIGN KEY (id_usuario) REFERENCES TUsuarios(id_usuario),
    FOREIGN KEY (id_rol) REFERENCES TRoles(id_rol)
);

-- ============================================================
-- 2. CLIENTES Y VEHICULOS
-- ============================================================

CREATE TABLE TClientes (
    id_cliente INT PRIMARY KEY AUTO_INCREMENT,
    -- Datos de persona (antes en TPersonas)
    ci VARCHAR(255),
    primer_nombre VARCHAR(255),
    segundo_nombre VARCHAR(255),
    apellido_paterno VARCHAR(255),
    apellido_materno VARCHAR(255),
    fecha_nacimiento DATE,
    telefono VARCHAR(255),
    -- Datos de cliente
    nit VARCHAR(255),
    direccion VARCHAR(255),

    estadoA BOOLEAN DEFAULT TRUE,
    usuarioA INT,
    fechahoraA DATETIME DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE TMotocicletas (
    id_motocicleta INT PRIMARY KEY AUTO_INCREMENT,
    id_cliente INT,
    placa VARCHAR(255),
    marca VARCHAR(255),
    modelo VARCHAR(255),
    anio INT,
    -- Campos de ficha técnica fusionados (TFichasTecnicas eliminada)
    nro_chasis VARCHAR(255),
    nro_motor VARCHAR(255),
    color VARCHAR(255),
    cilindrada VARCHAR(255),

    estadoA BOOLEAN DEFAULT TRUE,
    usuarioA INT,
    fechahoraA DATETIME DEFAULT CURRENT_TIMESTAMP,

    FOREIGN KEY (id_cliente) REFERENCES TClientes(id_cliente)
);

-- ============================================================
-- 3. TALLER Y REPARACIONES
-- ============================================================

CREATE TABLE TServicios (
    id_servicio INT PRIMARY KEY AUTO_INCREMENT,
    nombre VARCHAR(255) NOT NULL,
    descripcion TEXT,
    precio_estimado DECIMAL,

    estadoA BOOLEAN DEFAULT TRUE,
    usuarioA INT,
    fechahoraA DATETIME DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE TOrdenesTrabajo (
    id_orden INT PRIMARY KEY AUTO_INCREMENT,
    id_motocicleta INT,
    id_empleado INT,
    nro_orden VARCHAR(255),
    fecha_ingreso DATE,
    fecha_cierre DATE,
    estado VARCHAR(255),
    condicion_entrada TEXT,

    estadoA BOOLEAN DEFAULT TRUE,
    usuarioA INT,
    fechahoraA DATETIME DEFAULT CURRENT_TIMESTAMP,

    FOREIGN KEY (id_motocicleta) REFERENCES TMotocicletas(id_motocicleta),
    FOREIGN KEY (id_empleado) REFERENCES TEmpleados(id_empleado)
);

CREATE TABLE THistorialesEstado (
    id_historial INT PRIMARY KEY AUTO_INCREMENT,
    id_orden INT,
    fecha_hora_cambio DATETIME,
    estado_asignado VARCHAR(255),
    observacion TEXT,

    estadoA BOOLEAN DEFAULT TRUE,
    usuarioA INT,
    fechahoraA DATETIME DEFAULT CURRENT_TIMESTAMP,

    FOREIGN KEY (id_orden) REFERENCES TOrdenesTrabajo(id_orden)
);

CREATE TABLE TDetallesOrdenTrabajo (
    id_detalle_ot INT PRIMARY KEY AUTO_INCREMENT,
    id_orden INT,
    id_servicio INT,
    descripcion VARCHAR(255),
    cantidad INT,
    precio_labor DECIMAL,
    subtotal DECIMAL,

    estadoA BOOLEAN DEFAULT TRUE,
    usuarioA INT,
    fechahoraA DATETIME DEFAULT CURRENT_TIMESTAMP,

    FOREIGN KEY (id_orden) REFERENCES TOrdenesTrabajo(id_orden),
    FOREIGN KEY (id_servicio) REFERENCES TServicios(id_servicio)
);

CREATE TABLE TListasVerificacion (
    id_lista INT PRIMARY KEY AUTO_INCREMENT,
    id_orden INT UNIQUE,
    frenos_revisados BOOLEAN,
    luces_revisadas BOOLEAN,
    piezas_ajustadas BOOLEAN,
    prueba_ruta BOOLEAN,
    fecha_validacion DATE,
    fecha DATETIME,
    kilometraje INT,

    estadoA BOOLEAN DEFAULT TRUE,
    usuarioA INT,
    fechahoraA DATETIME DEFAULT CURRENT_TIMESTAMP,

    FOREIGN KEY (id_orden) REFERENCES TOrdenesTrabajo(id_orden)
);

-- 9. AUDITORIA

CREATE TABLE TAuditoriaGeneral (
    idAuditoria INT PRIMARY KEY AUTO_INCREMENT,
    TablaNombre VARCHAR(100) NOT NULL,
    RegistroId INT NOT NULL,
    Accion CHAR(1) NOT NULL COMMENT 'I=Insert, U=Update, D=Delete',
    Campo VARCHAR(100),
    ValorAnterior TEXT,
    ValorNuevo TEXT,
    usuarioA INT NOT NULL,
    fechaHoraA DATETIME DEFAULT CURRENT_TIMESTAMP,
    direccionIP VARCHAR(45),
    Detalles TEXT,

    FOREIGN KEY (usuarioA) REFERENCES TUsuarios(id_usuario)
);
